% PGFPlots configuration
\pgfplotsset{compat=1.18}
\pgfplotsset{
    every axis/.append style={
        ylabel near ticks,
        label style={font=\scriptsize},
        tick label style={font=\scriptsize},
        title style={font=\scriptsize},
        grid=both,
        grid style={dashed, color=lightgray},
        legend style={font=\scriptsize, nodes={scale=0.9}},
        legend image post style={scale=0.5},
        try min ticks = 6,
        max space between ticks = \ifx\fullver\undefined 15pt \else 16pt \fi,
    },
    discard if not/.style 2 args={
        x filter/.append code={
            \edef\tempa{\thisrow{#1}}
            \edef\tempb{#2}
            \ifx\tempa\tempb
            \else
                \def\pgfmathresult{}
            \fi
        }
    },
    select coords/.style n args={4}{
        x filter/.append code={
            \ifnum\coordindex=#1
            \else
                \ifnum\coordindex=#2
                \else
                    \ifnum\coordindex=#3
                    \else
                        \ifnum\coordindex=#4
                        \else
                            \def\pgfmathresult{}
                        \fi
                    \fi
                \fi
            \fi
        }
    },
}
